02. 브라우저 객체 : window 객체 메서드 : window.confirm() : 확인창 표시하기

클릭 : 확인창을 표시합니다. 
{
    const btn = document.querySelector(".btn2-1");

    btn.addEventListener("click", () => {
        const text = confirm("오늘 공부할 준비가 되었나요?");

        //document.querySelector(".result2").innerHTML = con;

        if( text ) {
            document.querySelector(".result2").innerHTML = "오늘 공부할 준비가 되었습니다.";
        } else {
            document.querySelector(".result2").innerHTML = "오늘 몸이 안 좋아요!!";
        }
    });
}
