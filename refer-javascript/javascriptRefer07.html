01. 날짜 메서드

번호 기본값 메서드 결과값 
{
    const date = new Date();
    const year = date.getFullYear();
    document.querySelector(".sample01_R1").innerHTML = "지금은 " + year + "년 입니다.";

    const month = date.getMonth() + 1;
    document.querySelector(".sample01_R2").innerHTML = "지금은 " + month + "월 입니다.";

    const days = date.getDate();
    document.querySelector(".sample01_R3").innerHTML = "지금은 " + days + "일 입니다.";

    const hour = date.getHours();
    document.querySelector(".sample01_R4").innerHTML = "지금은 " + hour + "시 입니다.";

    const minutes = date.getMinutes();
    document.querySelector(".sample01_R5").innerHTML = "지금은 " + minutes + "분 입니다.";

    const seconds = date.getSeconds();
    document.querySelector(".sample01_R6").innerHTML = "지금은 " + seconds + "초 입니다.";

    const milliseconds = date.getMilliseconds();
    document.querySelector(".sample01_R7").innerHTML = "지금은 " + milliseconds + "밀리초 입니다.";

    const day = date.getDay();
    const dayList = ['일요일', '월요일', '화요일', '수요일', '목요일', '금요일', '토요일'];
    document.querySelector(".sample01_R8").innerHTML = "지금은 " + dayList[day] + " 입니다.";
}

02. 날짜 메서드

번호 기본값 메서드 결과값 
{
    const date = new Date();
    const locale = date.toLocaleString();
    document.querySelector(".sample02_R1").innerHTML = locale;

    const localeDate = date.toLocaleDateString();
    document.querySelector(".sample02_R2").innerHTML = localeDate;

    const localeTime = date.toLocaleTimeString();
    document.querySelector(".sample02_R3").innerHTML = localeTime;
}
