01. 변수 : 데이터 저장소 

{
    var x = 100;            //변수 x에 숫자 100을 저장.
    var y = 200;            //변수 y에 숫자 200을 저장.
    var z = "javascript";   //변수 z에 문자 "javascript"를 저장.

    document.write("*** 01. 변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
    document.write(x + y);
    document.write(x * y);
}

02. 변수 : 변하는 데이터 저장소, 데이터 추가 

{
    let x = 100;
    let y = 200;
    let z = "javascript";
    x = 300;
    y = 400;
    z += "jquery";

    document.write("*** 02. 변수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}

03. 상수 : 데이터 저장소, 변하지 않는 데이터 

{
    const x = 100;
    const y = 200;
    const z = "javascript";
    //x = 200;
    //y = 300;

    document.write("*** 03. 상수 ***");
    document.write(x);
    document.write(y);
    document.write(z);
}
