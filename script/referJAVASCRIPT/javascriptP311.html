02. 수학 메서드 : Math.random()

번호 기본값 메서드 결과값 클릭 
{
    document.querySelector(".sample02_B1 button").addEventListener("click", mathRandom1);
    function mathRandom1(){
        document.querySelector(".sample02_R1").innerHTML = Math.random();
    }

    document.querySelector(".sample02_B2 button").addEventListener("click", mathRandom2);
    function mathRandom2(){
        document.querySelector(".sample02_R2").innerHTML = Math.floor(Math.random() * 10);
    }

    document.querySelector(".sample02_B3 button").addEventListener("click", mathRandom3);
    function mathRandom3(){
        document.querySelector(".sample02_R3").innerHTML = 10 + Math.floor(Math.random() * 10);
    }
}
